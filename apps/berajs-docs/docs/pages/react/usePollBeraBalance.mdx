# usePollBeraBalance

React hook to retrieve the current Bera balance of a given token.

## Import

```ts
import { usePollBeraBalance } from "@bera/berajs";
```

## Usage

:::code-group

```ts twoslash [example.ts]
import { usePollBeraBalance } from "@bera/berajs";

import { beraConfig } from "./config";

const exampleTokenAddress = "0x1";
const { useBalance, isLoading, isValidating } = useTokenHoneyPrice({
  args: { tokenAddress: exampleTokenAddress },
});

const { data } = useBalance();
```

```ts twoslash [config.ts] filename="config.ts"
import { BeraConfig } from "@bera/berajs";

// required configs for getHoneyBalance
export const beraConfig: BeraConfig = {
  contracts: {
    honeyAddress: "0x....",
  },
};
```

:::
