# useTokenInformation

React hook for grabbing token information.

## Import

```ts
import { useTokenInformation } from "@bera/berajs";
```

## Usage

:::code-group

```ts twoslash [example.ts]
import { useTokenInformation } from "@bera/berajs";

import { beraConfig } from "./config";

const {
  isLoading, // boolean indicating if token information is still loading
  isSuccess,
  isError,
  tokenInformation,
  error,
  read,
} = useTokenInformation();

const token = {
  address: "0x...",
  name: "...",
  symbol: "...",
  decimals: 18,
};
```

```ts twoslash [config.ts] filename="config.ts"
import { BeraConfig } from "@bera/berajs";

// required configs for getRoute
export const beraConfig: BeraConfig = {
  endpoints: {
    tokenList: "https://..",
  },
};
```

:::

## Parameters

```ts
import { type DefaultHookTypes } from "@bera/berajs";
```

- config.endpoints.tokenList
  - Type: `string`
  - Description: url to a [token list](https://tokenlists.org/) compatible json file
