# usePollAssetWalletBalance

React hook for polling balance of a token list for a connected user

## Import 

```ts
import { usePollAssetWalletBalance } from '@bera/berajs'
```

## Usage

:::code-group
```ts twoslash [example.ts]
import { usePollAssetWalletBalance } from '@bera/berajs'
import { beraConfig } from './config'

const {
    useCurrentAssetWalletBalances,
    useSelectedAssetWalletBalance,
    useSelectedTagAssetWalletBalance,
    refetch,
    isLoading
} = usePollAssetWalletBalance()

// list of all balances
const balances = useCurrentAssetWalletBalances()
// balance of selected asset
const selectedBalance = useSelectedAssetWalletBalance('0x...')
// balance of selected tag assets
const selectedTagBalance = useSelectedTagAssetWalletBalance('atoken')

```
