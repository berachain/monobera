# usePollHoneyBalance

React hook to retrieve the current Honey balance of the connected account.

## Import

```ts twoslash
import { usePollHoneyBalance } from "@bera/berajs";
```

## Usage

```ts twoslash [example.ts]
import { usePollHoneyBalance } from "@bera/berajs";
import { BeraConfig } from "@bera/berajs";

// required configs for getHoneyBalance
export const beraConfig: BeraConfig = {
  contracts: {
    honeyAddress: "0x....",
  },
};
  
const { data: honeyBalance, refetch } =
  usePollHoneyBalance({
    config: beraConfig,
  });
  
```
