# usePollHoneyBalance

React hook to retrieve the current honey balance of an address.

## Import

```ts
import { usePollHoneyBalance } from "@bera/berajs";
```

## Usage

:::code-group

```ts twoslash [example.ts]
import { usePollHoneyBalance } from "@bera/berajs";

import { beraConfig } from "./config";

const { useHoneyBalance, useRawHoneyBalance, isLoading, isValidating } =
  usePollHoneyBalance({
    config: beraConfig,
  });

const { data: formattedHoneyPrice } = useHoneyBalance();
const { data: rawHoneyPrice } = useRawHoneyBalance();
```

```ts twoslash [config.ts] filename="config.ts"
import { BeraConfig } from "@bera/berajs";

// required configs for getHoneyBalance
export const beraConfig: BeraConfig = {
  contracts: {
    honeyAddress: "0x....",
  },
};
```

:::
