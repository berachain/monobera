# useTokenHoneyPrice

React hook to retrieve the current honey prices of a series of tokens

## Import

```ts
import { useTokenHoneyPrices } from "@bera/berajs";
```

## Usage

:::code-group

```ts twoslash [example.ts]
import { useTokenHoneyPrices, type Token } from "@bera/berajs";

import { beraConfig } from "./config";

const exampleTokenAddresses = ["0x1", "0x2"];
const { data: tokePricesInHoney, isLoading } = useTokenHoneyPrices({
  config: beraConfig,
  args: { tokenAddresses: exampleTokenAddresses },
});
```

```ts twoslash [config.ts] filename="config.ts"
import { BeraConfig } from "@bera/berajs";

// required configs for getTokenHoneyPrices
export const beraConfig: BeraConfig = {
  subgraphs: {
    dexSubgraph: "https://...",
  },
};
```

:::

## Parameters

```ts
import { type UseTokenHoneyPricesRequest } from "@bera/berajs";
```

- args.tokenAddresses
  - Type: `string[]`
  - Description: Addresses of the given tokens
