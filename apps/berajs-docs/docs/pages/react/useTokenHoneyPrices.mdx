# useTokenHoneyPrice

React hook to retrieve the current honey prices of a series of tokens

## Import

```ts twoslash
import { useTokenHoneyPrices } from "@bera/berajs";
```

## Usage

```ts twoslash [example.ts]
import { Token, useTokenHoneyPrices } from "@bera/berajs";

import { beraConfig } from "./config";

const { data: tokenPricesInHoney, refresh } = useTokenHoneyPrices({
  config: beraConfig,
  args: { tokenAddresses: ["0x0...", "0x1..."] },
});
```

```ts twoslash [config.ts] filename="config.ts"
import { BeraConfig } from "@bera/berajs";

// required configs for getTokenHoneyPrices
export const beraConfig: BeraConfig = {
  subgraphs: {
    dexSubgraph: "https://...",
  },
};

const { data: tokenPricesInHoney, refetch } = useTokenHoneyPrices(
  {
    tokenAddresses: ["0x0...", "0x1..."],
  },
  {
    beraConfigOverride: beraConfig, // optional arg to override default beraConfig
  },
);
```
