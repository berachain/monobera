# usePollTransactionCount

React hook for polling transaction counts for a connected user

## Import

```ts
import { usePollTransactionCount } from "@bera/berajs";
```

## Usage

```ts [example.ts]
import { usePollWalletBalances } from "@bera/berajs";

import { beraConfig } from "./config";

const exampleAddress = "0x1..";

const { useTransactionCount } = usePollTransactionCount({
  config: beraConfig,
  args: {
    address: exampleAddress,
  },
});

// number of transactions
const txCounts = useTransactionCount();
```

:::
