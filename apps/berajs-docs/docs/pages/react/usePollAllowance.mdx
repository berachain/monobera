# usePollAllowance

React hook for poll the allowance of a given token

## Import

```ts twoslash
import { usePollAllowance } from "@bera/berajs";
```

## Usage

```ts twoslash [example.ts]
import { usePollAllowance, type Token } from "@bera/berajs";

const exampleToken: Token = {
  address: "0x0",
  decimals: 18,
  symbol: "TKN",
  name: "TKN",
};
const { data: allowance, refresh } = usePollAllowance({
  spender: "0x0...",
  token: exampleToken,
});
```
