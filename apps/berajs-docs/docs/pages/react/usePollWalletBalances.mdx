# usePollWalletBalances

React hook for polling balance of a token list for a connected user

## Import 

```ts
import { usePollWalletBalances } from '@bera/berajs'
```

## Usage

:::code-group
```ts twoslash [example.ts]
import { usePollWalletBalances } from '@bera/berajs'
import { beraConfig } from './config'

const {
    useCurrentWalletBalances,
    useSelectedWalletBalance,
    useSelectedTagWalletBalances,
    refetch,
    isLoading
} = usePollWalletBalances(beraConfig)

// list of all balances
const balances = useCurrentWalletBalances()
// balance of selected asset
const selectedBalance = useSelectedWalletBalance('0x...')
// balance of selected tag assets
const selectedTagBalance = useSelectedTagWalletBalances('atoken')

```

```ts twoslash [config.ts] filename="config.ts"
import { BeraConfig } from "@bera/berajs";

// required configs for getRoute
export const beraConfig: BeraConfig = {
    endpoints: {
        tokenList: 'https://..',
    }
}
```
:::


## Parameters
```ts twoslash
import { type UsePollWalletBalancesRequest } from '@bera/berajs'
```

- config.endpoints.tokenList
    - Type: `string`
    - Description: url to a [token list](https://tokenlists.org/) compatible json file
