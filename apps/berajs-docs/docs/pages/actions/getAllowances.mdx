# getAllowances

Action for fetching the allowances of a given token array

## Initialization

Ensure you have followed the steps from [Actions > Getting Started](/actions/getting-started) before getting started

## Import

```ts
import { getAllowances } from "@bera/berajs";
```

## Usage

```ts
import { getAllowances } from "@bera/berajs";
import { createPublicClient, http } from "viem";
import { berachain } from "viem/chains";
import { usePublicClient } from "wagmi";

// first, intialize a public client with berachain
const publicClient = createPublicClient({
  chain: berachain,
  transport: http(),
});

interface AllowanceToken extends Token {
  allowance: bigint;
  formattedAllowance: string;
}

interface Call {
  abi: typeof erc20Abi;
  address: `0x${string}`;
  functionName: string;
  args: any[];
}

const { data } = await getAllowance({
  tokens,
  account,
  config,
  contract,
  publicClient,
  queryKey,
  mutate,
});
```
