# getRoute

Action for fetching the optimal swap route between two tokens.

## Import

```ts
import { getRoute } from "@bera/berajs/actions";
```

## Required config

For more information on BeraConfig, see [BeraConfig](/bera-config/)

```ts twoslash
import { BeraConfig } from "@bera/berajs";

// required configs for getRoute
export const beraConfig: BeraConfig = {
  endpoints: {
    dexRouter: "https://...",
  },
};
```

## Usage

```ts twoslash
import { SwapInfoV3 } from "@bera/berajs";
import { getRoute } from "@bera/berajs/actions";

import { beraConfig } from "./config";

const route = await getRoute({
  args: {
    tokenIn: "0x1",
    tokenOut: "0x2",
    tokenInDecimals: 18,
    tokenOutDecimals: 18,
    amount: "1.1",
  },
  config: beraConfig,
});
```
