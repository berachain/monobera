# getTokenInformation

Action for fetching token information

## Initialization

Ensure you have followed the steps from [Actions > Getting Started](/actions/getting-started) before getting started

## Import

```ts
import { getTokenInformation } from "@bera/berajs/actions";
```

## Usage

:::code

```ts [example.ts]
import { type BeraConfig } from "@bera/berajs";
import { getTokenInformation } from "@bera/berajs/actions";
import { createPublicClient, http } from "viem";
import { berachain } from "viem/chains";

// first, intialize a public client with berachain
const publicClient = createPublicClient({
  chain: berachain, // viem chains directory: https://github.com/wevm/viem/blob/main/src/chains/index.ts
  transport: http(),
});

// required configs for getWalletBalances
export const beraConfig: BeraConfig = {
  contracts: {
    multicallAddress: "0x...",
  },
};

await getTokenInformation({
  address,
  config: beraConfig,
  publicClient,
});
```

:::

## Parameters

| Name         | type         | Description                                                             | Required |
| ------------ | :----------- | :---------------------------------------------------------------------- | :------: |
| address      | `Address`    | Blockchain address of the token                                         |  false   |
| config       | `BeraConfig` | Configuration settings specific to the token retrieval                  |  false   |
| publicClient | `any`        | Client interface for public blockchain interactions, type not specified |  false   |

## Returns

#### [`Promise<Token | undefined>`](/types/Token/)
