# getWalletBalances

Action for fetching the balances of a given wallet address

## Initialization

Ensure you have followed the steps from [Actions > Getting Started](/actions/getting-started) before getting started

## Import

```ts
import { getWalletBalances } from "@bera/berajs/actions";
```

## Usage

```ts
import { type BeraConfig } from "@bera/berajs";
import { getWalletBalances, type Token } from "@bera/berajs/actions";
import { createPublicClient, http } from "viem";
import { berachain } from "viem/chains";

// first, intialize a public client with berachain
const publicClient = createPublicClient({
  chain: berachain, // viem chains directory: https://github.com/wevm/viem/blob/main/src/chains/index.ts
  transport: http(),
});

// required configs for getWalletBalances
export const beraConfig: BeraConfig = {
  endpoints: {
    tokenList: "https://..",
  },
};

const walletBalances = await getWalletBalances({
  account,
  tokenList,
  config: beraConfig,
  publicClient,
});
```
