# getAllowances

Action for fetching the allowances of a given token array

## Initialization

Ensure you have followed the steps from [Actions > Getting Started](/actions/getting-started) before getting started

## Import

```ts
import { getAllowances } from "@bera/berajs/actions";
```

## Usage

```ts
import { getAllowances } from "@bera/berajs/actions";
import { createPublicClient, http } from "viem";
import { berachain } from "viem/chains";
import { usePublicClient } from "wagmi";

// first, intialize a public client with berachain
const publicClient = createPublicClient({
  chain: berachain, // viem chains directory: https://github.com/wevm/viem/blob/main/src/chains/index.ts
  transport: http(),
});

interface AllowanceToken extends Token {
  allowance: bigint;
  formattedAllowance: string;
}

interface Call {
  abi: typeof erc20Abi;
  address: `0x${string}`;
  functionName: string;
  args: any[];
}

const { data } = await getAllowance({
  tokens,
  account,
  config,
  spender,
  publicClient,
});
```


## Parameters

| Name           | type                              | Description                                                               | Required |
| -------------- | :-------------------------------- | :------------------------------------------------------------------------ | :------: |
| account      | `Address` \| `undefined`  | Blockchain address of the account holder         | true     |
| token        | `Token` \| `undefined`    | Reference to a `Token` object or undefined       | true     |
| spender      | `Address`                 | Blockchain address of the entity spending the token | false    |
| publicClient | `PublicClient` \| `undefined` | Client interface for public blockchain interactions | true     |

## Returns

#### [`AllowanceToken | undefined`](/types/AllowanceToken/)
