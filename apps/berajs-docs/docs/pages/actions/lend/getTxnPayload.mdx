# getTxnPayload

Action for creating the payload for transaction in lend.

## Import

```ts
import {
  getBorrowPayload,
  getRepayPayload,
  getSupplyPayload,
  getWithdrawPayload,
} from "@bera/berajs";
```

## Usage

```ts
getBorrowPayload({ token, amount, account });

getSupplyPayload({ token, amount, account });

getRepayPayload({
  token,
  amount,
  repayMax,
  account,
});

getWithdrawPayload({
  token,
  amount,
  withdrawMax,
  account,
});

```

## Parameters and Returns

```ts
declare const getSupplyPayload: ({ token, amount, account, }: {
    token: Token;
    amount: string;
    account: Address;
}) => (number | bigint | `0x${string}`)[];

declare const getWithdrawPayload: ({ token, amount, withdrawMax, account, }: {
    token: Token;
    amount: string;
    withdrawMax?: boolean;
    account: Address;
}) => (bigint | `0x${string}`)[];

declare const getBorrowPayload: ({ token, amount, account, }: {
    token: Token;
    amount: string;
    account: Address;
}) => (number | bigint | `0x${string}`)[];

declare const getRepayPayload: ({ token, amount, repayMax, account, }: {
    token: Token;
    amount: string;
    repayMax?: boolean;
    account: Address;
}) => (number | bigint | `0x${string}`)[];
```
