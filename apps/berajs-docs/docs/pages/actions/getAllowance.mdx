# getAllowance

Action for fetching the allowance of a given token

## Initialization

Ensure you have followed the steps from [Actions > Getting Started](/actions/getting-started) before getting started

## Import

```ts
import { getAllowance } from "@bera/berajs";
```

## Usage

```ts
import { getAllowance } from "@bera/berajs";
import { createPublicClient, http } from "viem";
import { berachain } from "viem/chains";

// first, intialize a public client with berachain
const publicClient = createPublicClient({
  chain: berachain,
  transport: http(),
});

const { token, allowance, formattedAllowance } = await getAllowance({
  account,
  token,
  contract,
  publicClient,
});
```

import { type IGetAllowanceRequest } from "berajs";
