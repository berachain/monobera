# getAllowance

Action for fetching the allowance of a given token

## Import

```ts
import { getAllowance } from "@bera/berajs/actions";
```

## Usage

```ts
import { getAllowance } from "@bera/berajs/actions";
import { usePublicClient } from "wagmi";

const client = usePublicClient();

const { token, allowance, formattedAllowance } = await getAllowance({
  account,
  token,
  contract,
  publicClient: client,
});
```

## Parameters

```ts
import { type GetAllowance } from "@bera/berajs/actions";
```

## Returns

```ts
import { type GetAllowanceResponse } from "@bera/berajs/actions";

type result = Promise<GetAllowanceResponse>;
```
