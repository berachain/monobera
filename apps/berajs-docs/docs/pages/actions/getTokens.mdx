# getTokens

Action for fetching and formatting the token list

## Initialization

Ensure you have followed the steps from [Actions > Getting Started](/actions/getting-started) before getting started

## Import

```ts
import { getTokens } from "@bera/berajs";
```

## Usage

```ts
import { getTokens, type Token } from "@bera/berajs";
import { useLocalStorage } from "usehooks-ts";

const TOKEN_KEY = "tokens";

const [localStorageTokenList, setLocalStorageTokenList] = useLocalStorage<
  Token[]
>(TOKEN_KEY, []);

const tokens = await getTokens({
  localStorageTokenList,
});
```

## Parameters

```ts
import { type IFetchTokensRequestArgs } from "@bera/berajs";
```

## Returns

```ts
import { type IFetchTokensResponse } from "@bera/berajs";

type result = Promise<IFetchTokensResponse | Token[] | undefined>;
```
