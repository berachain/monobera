# getTokens

Action for fetching and formatting the token list

## Import

```ts
import { getTokens } from "@bera/berajs/actions";
```

## Usage

```ts
import { getTokens, type Token } from "@bera/berajs/actions";

import { beraConfig } from "./config";

const customToken: Token = {
  address: "0x0...",
  decimals: 18,
  symbol: "TKN",
  name: "TKN",
}

const tokens = await getTokens({
  externalList: [customToken], // optional list of Token Objects to combine with the token list in the config
  config: beraConfig,
});
```

## Parameters

```ts 
import { type GetTokensRequest } from "@bera/berajs/actions";
```

import { type GetTokens } from "@bera/berajs/actions";
